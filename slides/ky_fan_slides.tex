\begin{frame}{Неспектральные матричные нормы для обучения нейронных сетей}
    \begin{block}{Проблема}
        Поиск \coloreddef{$\argmin_{\mX\in\R^{m\times n}} f(\mX)$}, в особенности когда $\mX$ — матрица весов в нейронной сети, а $f$ невыпуклая.
    \end{block}

    \begin{block}{Цель}
        В рамках подхода \coloreddef{norm-constrained linear minimization oracle} (LMO) \citep{pethick2025training} \coloreddef{отойти от спектральной нормы}, чтобы получить эффективные алгоритмы для обучения внутренних слоёв нейросети взамен Muon.
    \end{block}

    \begin{block}{Решение}
        На основе норм Ки Фана предлагается построить матричные нормы, порождающие семейства алгоритмов \emph{Fanions}, \coloredresult{связанные с Muon, $\nu$-SAM и Dion,} а также \emph{F-Fanions} и \emph{S-Fanions}, объединяющие Fanions с \coloredresult{Normalized SGD (NSGD) и SignSGD}.
    \end{block}
\end{frame}

%-------------------------------------------------------------------------------------
\begin{frame}{Norm-constrained LMO: задача выбора нормы}
  Наделим пространство $\Rmn$ нормой $\norm{\cdot}$ с единичным шаром $\cB_1 = \{ \mD \in \Rmn \mid \norm{\mD} \le 1 \}$. Её двойственная норма определяется как \coloreddef{$\norm{\mM}^\dagger = \sup_{\mD \in \cB_1} \langle \mM, \mD \rangle.$}

  \vspace{0.4em}

  Направление \coloreddef{$\mathrm{LMO}(\mM^t) \in \Argmin_{\mD \in \cB_1} \langle\mM^t, \mD\rangle$}, где $\mM^t$ — градиент или моментум, задаёт обновление:
  \[
  \begin{aligned}
    \mX^{t+1} &= \mX^{t} + \gamma_t \, \mathrm{LMO}(\mM^t) \in \mX^{t} - \gamma_t \Argmax_{\mD \in \cB_1} \langle \mM^t, \mD \rangle \\
    &= \coloredupdate{\mX^{t} - \gamma_t \{\Delta \in \cB_1 \mid \langle \mM^t, \Delta \rangle = \norm{\mM^t}^\dagger\}}.
    \end{aligned}
  \]
  Достаточно \coloredresult{найти направление $\Delta \in \cB_1$}, которое даёт равенство \coloredresult{$\<\mM^t, \Delta> = \norm{\mM^t}^\dagger$}. Для этого часто требуется SVD разложение: $\mM^t = \mU \mSigma \mV^\top = \mU \diag(\sigma_1, \dots, \sigma_r) \mV^\top$.
  \begin{block}{Требуется}
    Выяснить, какие $\norm{\cdot}$, помимо спектральной $\norms{\cdot}$, задают эффективные на практике алгоритмы.
 \end{block}
\end{frame}

\begin{frame}{Проблема ранга LMO-обновлений}
  Нормы Шаттена \coloreddef{$\norm{\mM^t}_{S_p} = \left(\sum_{i=1}^{\min(m, n)}\sigma_i^p\right)^{1/p}\,$} дают шаг обновления
  \[\coloredupdate{
  \mX^{t+1} = \mX^t - \gamma_t \mU \frac{\diag\left(\sigma_1^{q-1},\dots, \sigma_{\min(m,n)}^{q-1}\right)}{\left(\sum_{i=1}^{\min(m,n)}{\sigma_i^q}\right)^{\frac{q-1}{q}}}\mV^\top}
  \]
  где $p^{-1} + q^{-1} = 1$ \citep{cesista2025schattenp}.

  \vspace{0.5em}
  Нормы Ки Фана \coloreddef{$\normkfk{\mM^t}= \sum_{i=1}^{k}\sigma_i\,$} дают 
  \[\coloredupdate{
  \mX^{t+1} = \mX^{t} - \gamma_t \vu_1 \vv_1^\top\text{ \quad или \quad }\mX^{t+1} = \mX^{t} - \frac{\gamma_t}{k}\mU \mV^\top\,,}
  \]
  что соответствует алгоритмам Neon или Muon \citep{kravatskiy2025fanions}.

  \vspace{0.5em}
  \alert{Следствие:} обновления получились либо одноранговыми, либо полноранговыми.
\end{frame}

%----------------------------------------------------------------------------------------------------------
\begin{frame}{Нормы $\normkfk{\cdot}^\dagger$, двойственные к нормам Ки Фана, и \coloreddef{алгоритмы Fanions}}
  Пусть \coloreddef{$\norm{\cdot} = \normkfk{\cdot}^\dagger$}. Возьмём направление $\Delta = \sum_{i=1}^k{\vu_i \vv_i^\top}$. Его норма $\normkfk{\Delta}^\dagger = \max\{\frac{1}{k} \normn{\Delta}, \norms{\Delta}\} = \max\{\frac{1}{k} k, 1\} = 1$, и скалярное произведение 
  \[
  \begin{aligned}
  \<\mM^t, \Delta> &=\<\mU \mSigma \mV^\top, \sum_{i=1}^{k} \vu_i \vv_i^\top> = \sum_{i,j=1}^{r, k}\<\vu_i \sigma_i \vv_i^\top, \vu_j \vv_j^\top> = \sum_{i=1}^{k}\sigma_i \\
  &= \normkfk{\mM^t} = \normkfk{\mM^t}^{\dagger\dagger} = \norm{\mM^t}^\dagger.
  \end{aligned}
  \]  
  Следовательно, искомое обновление для Fanion-$k$ принимает вид:
  $$\coloredupdate{\mX^{t+1} = \mX^{t} - \gamma_t \sum_{i=1}^{k} \vu_i \vv_i^\top}$$
\end{frame}

\begin{frame}{Коническая комбинация LMO}
    \begin{block}{Лемма о конической комбинации LMO алгоритмов (Кравацкий, 2025)}
      Пусть задано конечное семейство LMO-алгоритмов $i = 1,\dots,n$, где обновление $i$-го алгоритма на шаге $t$ определяется как
      \[
      \mX^{t + 1} - \mX^{t} = \gamma_t \mathrm{LMO}_{\norm{\cdot}_{(i)}}(\mM^{t})\,,
      \]
      и $\mathrm{LMO}_{\norm{\cdot}_{(i)}}$ соответствует единичному шару нормы $\norm{\cdot}_{(i)}$.
      
      Тогда для любых неотрицательных $\alpha_1,\dots,\alpha_n$, алгоритм с обновлением
      \[
      \coloredupdate{\mX^{t + 1} - \mX^{t} = \gamma_t \sum_{i = 1}^n \alpha_i \mathrm{LMO}_{\norm{\cdot}_{(i)}}(\mM^t)}
      \]
      сам \coloredresult{является LMO-алгоритмом}, соответствующим единичному шару \coloredresult{нормы, двойственной к норме $\sum_{i = 1}^n \alpha_i \norm{\cdot}_{(i)}^\dagger$}.
    \end{block}
\end{frame}

\begin{frame}{Гибридные алгоритмы: F-Fanions и S-Fanions}

  $\alpha \times$Fanion (норма $\normkfk{\cdot}^\dagger$) + $(1-\alpha) \times$NSGD (норма $\normf{\cdot}$) = \coloreddef{F-Fanion}:
  $$\coloredupdate{\mX^{t+1} = \mX^{t} - \gamma_t \left(\alpha \sum_{i=1}^k{\vu_i \vv_i^\top} + (1-\alpha)\frac{\mM^t}{\normf{\mM^t}}\right)}$$

  \vspace{0.5em}
  $\alpha \times$Fanion + $(1-\alpha)\times\eta\times$SignSGD (норма $\eta||\cdot||_\infty$) = \coloreddef{ S-Fanion}:  

  $$\coloredupdate{\mX^{t+1} = \mX^{t} - \gamma_t \left(\alpha \sum_{i = 1}^k \vu_i \vv_i^\top + (1-\alpha)\eta\sign(\mM^t)\right)}$$

\end{frame}
